     PROGRAMA PARA CONVERTIR PESOS A DOLARES
#include <stdio.h>
#include <stdlib.h>

/* run this program using the console pauser or add your own getch, system("pause") or input loop */

int main() {
float pesos, trm, cantidad;

printf("Ingrese la cantidad de pesos que desea convertir *Solo se admiten numeros\n");
scanf("%f",&pesos);
printf("Ingrese la trm del dia\n");
scanf("%f",&trm);
cantidad=pesos/trm;
printf("La cantidad de dolares es: %f\n", cantidad);

system("pause");
return 0;
}
