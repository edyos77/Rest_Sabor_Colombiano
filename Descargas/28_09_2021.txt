       PROGRAMA PARA CALCULAR EL IMC
#include <stdio.h>
#include <stdlib.h>
/* run this program using the console pauser or add your own getch, system("pause") or input loop */
int main() {

float estatura, peso, imc;
printf("Programa para calcular el indice de masa corporal (IMC)\n\n");
printf("Ingrese su estatura en metros: (ej. 1.65).  ");
scanf("%f",&estatura);
printf("Ingrese su peso en kilogramos: (ej. 70.5).  ");
scanf("%f", &peso);

imc=peso/(estatura*estatura);
if(imc<18.5){
printf("Usted tiene un Peso bajo.\n");
}
else if(imc<24.9){
printf("Usted tiene un Peso normal.\n");
}
else if(imc<29.9){
printf("Usted tiene Sobrepeso.\n");
}
else{
printf("Usted tiene Obesidad.\n");
}

system("pause");
return 0;
}

       PROGRAMA QUE MUESTRA UNA TABLA DE MULTIPLICAR
#include <stdio.h>
#include <stdlib.h>
/* run this program using the console pauser or add your own getch, system("pause") or input loop */
int main() {
printf("Programa que muestra una tabla de multiplicar\n\n ");
int tabla, numero, contador=1;
printf("Ingrese la tabla que desea consultar:  ");
scanf("%i", &numero);
while(contador<=10){
tabla=numero*contador;
printf("%i X %i = %i\n",numero, contador, tabla);
contador+=1;
}

system("pause");
return 0;
}
