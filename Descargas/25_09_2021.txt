       PROGRAMA QUE MUESTRA EN QUE ETAPA ESTA UNA PERSONA SEGUN SU EDAD
#include <stdio.h>
#include <stdlib.h>

/* run this program using the console pauser or add your own getch, system("pause") or input loop */

int main() {
int edad;

printf("Ingrese la edad: ");
scanf("%i",&edad);

if(edad>65){
printf("Adulto Mayor \n");
}
else if(edad>17){
printf("Adulto \n");
}
else if(edad>10){
printf("Adolescente \n");
}
else if(edad>6){
printf("Preadolescente \n");
}
else if(edad>2){
printf("Niño \n");
}
else{
printf("Bebé \n");
}

system("pause");
return 0;
}
